async function realizarLogin(e) { e.preventDefault(); const btn = document.querySelector('button[type="submit"]'); const textoOriginalBtn = btn.innerText; btn.innerHTML = 'Conectando...'; btn.disabled = true; // Pega os inputs (Adaptado para pegar os IDs mais comuns que usamos) const inputLogin = document.getElementById('loginUser') || document.getElementById('loginDoc'); const inputSenha = document.getElementById('loginPass') || document.getElementById('loginPwd'); // O .trim() mata os espaços em branco acidentais const rawUser = inputLogin.value.trim(); const rawPass = inputSenha.value.trim(); const soNumeros = rawUser.replace(/\D/g, ''); // 1. TENTA ACESSO SUPERADMIN MASTER if ((rawUser === 'admin' || soNumeros === '00000000000') && rawPass === 'admin123') { window.location.href = 'superadmin.html'; return; } try { // 2. TENTA ACESSO COMO DONO DA OFICINA (GESTOR) // Busca Dupla Inteligente: Procura por Usuário, se não achar, procura por CNPJ let ofcSnap = await db.collection('oficinas').where('usuario', '==', rawUser).where('senha', '==', rawPass).get(); if (ofcSnap.empty && soNumeros !== '') { ofcSnap = await db.collection('oficinas').where('cnpj', '==', soNumeros).where('senha', '==', rawPass).get(); } if (!ofcSnap.empty) { const ofc = ofcSnap.docs[0].data(); if (ofc.status === 'Bloqueado' || ofc.status === 'Inativo') { alert('Acesso Negado: A licença desta oficina encontra-se bloqueada.'); } else { // Grava a sessão do Gestor sessionStorage.setItem('t_id', ofcSnap.docs[0].id); sessionStorage.setItem('t_nome', ofc.nomeFantasia); sessionStorage.setItem('t_role', 'gerente'); // Herda chaves do Cloudinary se existirem no banco (Fallback de segurança) sessionStorage.setItem('t_cloudName', (ofc.apiKeys && ofc.apiKeys.cloudinaryName) ? ofc.apiKeys.cloudinaryName : 'dmuvm1o6m'); sessionStorage.setItem('t_cloudPreset', (ofc.apiKeys && ofc.apiKeys.cloudinaryPreset) ? ofc.apiKeys.cloudinaryPreset : 'evolution'); // Redireciona para o Painel da Oficina! window.location.href = 'painel_oficina.html'; return; // Para a execução aqui } } else { // 3. TENTA ACESSO COMO EQUIPE (MECÂNICO) const funcSnap = await db.collection('funcionarios').where('usuario', '==', rawUser).where('senha', '==', rawPass).get(); if (!funcSnap.empty) { const func = funcSnap.docs[0].data(); const funcId = funcSnap.docs[0].id; // Verifica se a oficina "mãe" dele não está bloqueada por falta de pagamento const maeSnap = await db.collection('oficinas').doc(func.tenantId).get(); if (!maeSnap.exists || maeSnap.data().status === 'Bloqueado' || maeSnap.data().status === 'Inativo') { alert('Acesso Negado: A oficina matriz está bloqueada no sistema.'); } else { // Grava a sessão do Mecânico sessionStorage.setItem('t_id', func.tenantId); sessionStorage.setItem('t_nome', maeSnap.data().nomeFantasia); sessionStorage.setItem('t_role', func.role || 'equipe'); // RBAC vai esconder o DRE sessionStorage.setItem('f_id', funcId); sessionStorage.setItem('f_nome', func.nome); sessionStorage.setItem('f_comissao', func.comissao_mo || 0); sessionStorage.setItem('f_comissao_pecas', func.comissao_pecas || 0); // Mecânico também vai pro painel, a arquitetura oculta as abas proibidas window.location.href = 'painel_oficina.html'; return; } } else { // Se não achou Gestor nem Mecânico alert('Credenciais não localizadas na base de dados. Verifique o usuário e a senha.'); } } } catch (err) { console.error("Erro na consulta do Login: ", err); alert('Erro estrutural ao conectar com a nuvem (Firebase).'); } // Se der erro, restaura o botão btn.innerHTML = textoOriginalBtn; btn.disabled = false; }
